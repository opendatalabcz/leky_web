Table DLP_LECIVEPRIPRAVKY {
  KOD_SUKL VARCHAR2(7) [primary key] // Kód léčivého přípravku (dále LP) přidělený SÚKL variantě LP v rámci rozhodnutí o registraci LP, přidělený neregistrovanému LP zařazenému do specifického léčebného programu (dále SLP) nebo přidělený potravině pro zvláštní lékařské účely (dále PZLÚ)

  H VARCHAR2(1) // Označení LP, který podléhá povinnému hlášení (povinnost předkládat ústavu před propuštěním na trh vzorky každé šarže přípravku k přezkoušení, § 32 odst. 4 písm.a) zákona o léčivech
  NAZEV VARCHAR2(70) // Název LP nebo PZLÚ
  SILA VARCHAR2(24) // Síla LP, kterou se rozumí obsah léčivých látek vyjádřený kvantitativně vzhledem k jednotce dávky, objemu nebo hmotnosti podle lékové formy
  FORMA VARCHAR2(27) // Léková forma
  BALENI VARCHAR2(22) // Velikost balení
  CESTA VARCHAR2(15) // Cesta podání
  DOPLNEK VARCHAR2(75) // Doplněk názvu LP, který jednoznačně určuje variantu LP, sestávající z integrace cesty podání LP, jeho lékové formy, velikosti balení a síly. Jedná se o starší položku číselníku, dále upřesněnou v položkách CESTA, FORMA, BALENÍ a SÍLA
  OBAL VARCHAR2(3) // Vnitřní obal LP, kterým se rozumí taková forma obalu, který je v bezprostředním kontaktu s LP

  DRZ VARCHAR2(4) // Držitel rozhodnutí o registraci
  ZEMDRZ VARCHAR2(3) // Země sídla držitele rozhodnutí o registraci
  AKT_DRZ VARCHAR2(3) // Aktuální držitel reg. rozhodnutí. Uvádí se jen u registrací B
  AKT_ZEM VARCHAR2(3) // Země aktuálního držitele rozhodnutí o reg.
  REG VARCHAR2(2) // Stav registrace
  V_PLATDO DATE // Platnost registrace do, není-li stanovena neomezená platnost podle § 34 zákona o léčivech. Uvádí se jen u registrací B, C, Y, F
  NEOMEZ VARCHAR2(1) // V poli vyplněno (X) v případě neomezené platnosti registrace (podle § 34 zákona o léčivech)

  UVADENIDO DATE // Uvádění na trh do, není-li stanovena neomezená platnost podle § 34 zákona o léčivech. Uvádí se jen u registrací B, C, Y

  IS_ NUMBER(2,0) // Indikační skupina
  ATC_WHO VARCHAR2(7) // Anatomicko-terapeuticko-chemická skupina (WHO Index)

  RC VARCHAR2(16) // Registrační číslo

  SDOV VARCHAR2(11) // Identifikační číslo souběžného dovozu, které se váže na odkazovaný referenční přípravek dle registračního čísla; zpravidla ve tvaru PI/xxx/yyyy
  SDOV_DOD VARCHAR2(4) // Zkratka dovozce u souběžného dovozu
  SDOV_ZEM VARCHAR2(3) // Země dovozce u souběžného dovozu

  REG_PROC VARCHAR2(3) // Registrační procedura

  DDDAMNT_WHO NUMBER(6,3) // Množství léčivé látky v definované denní dávce (DDD)
  DDDUN_WHO VARCHAR2(4) // Jednotka množství léčivé látky v DDD
  DDDP_WHO NUMBER(11,4) // Počet DDD v balení (na 4 desetinná místa)

  ZDROJ_WHO VARCHAR2(4) // WHO Index
  LL VARCHAR2(2000) // Seznam léčivých látek, oddělený čárkami

  VYDEJ VARCHAR2(1) // Klasifikace typu výdeje léčivého přípravku
  ZAV VARCHAR2(2) // Závislost - skupina omamných nebo psychotropních látek (návykové látky)
  DOPING VARCHAR2(7) // Doping
  NARVLA VARCHAR2(2) // Označení látky s anabolickým a jiným hormonálním účinkem podle nařízení vlády 454/2009 Sb.

  DODAVKY VARCHAR2(1) // Přípravek byl dodáván do lékáren a zdr. zařízení v posledních 6 měsících

  EAN VARCHAR2(13) // Pole je prázdné, údaje již nejsou k dispozici
  BRAILLOVO_PISMO VARCHAR2(1) // Braillovo písmo: schváleno (S), výjimka (V), nežádáno (mezera)

  EXP VARCHAR2(3) // Expirace
  EXP_T VARCHAR2(1) // Expirace typ: hodiny H, dny D, týdny W, měsíce M

  NAZEV_REG VARCHAR2(70) // Registrovaný název LP nebo PZLÚ
  MRP_CISLO VARCHAR2(13) // MRP číslo
  PRAVNI_ZAKLAD_REGISTRACE VARCHAR2(4) // Právní základ registrace

  OCHRANNY_PRVEK VARCHAR2(1) // Ochranný prvek, hodnota A - léčivý přípravek musí mít ochranný prvek
  OMEZENI_PRESKRIPCE_SMP VARCHAR2(1) // Omezení používání dle SmPC, hodnota A - léčivý přípravek má omezení používání dle SmPC ve smyslu §33 odst. 2 Vyhlášky č. 376/2011 Sb.
  TYP_LP VARCHAR2(10) // Typ léčivého přípravku
}
